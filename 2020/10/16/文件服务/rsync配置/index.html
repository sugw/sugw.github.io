安装

mkdir -p /nfsbackup
useradd rsync -M /sbin/nologin -s
chown -R rsync.rsync /nfsbackup

#配置文件
vim /etc/rsyncd.conf
###rsyncd.conf start##
uid = rsync
gid = rsync
use chroot = no
max connections = 200
timeout = 300
pid file = /var/run/rsyncd.pid
lock file = /var/run/rsync.lock
log file = /var/log/rsyncd.log
ignore errors
read only = false
list = false
hosts allow = 172.16.1.0/24
hosts deny = 0.0.0.0/32
auth users = rsync_backup
secrets file = /etc/rsync.password

[backup]
path = /backup
[nfsbackup]
path = /nfsbackup
##rsync_config__________end

密码文件
cat /etc/rsync.password
rsync_backup:suguowei
设置权限
chmod 600 /etc/rsync.password

启动rsync
systemctl start rsyncd
检查
ss -nltup|grep 873
nfs服务器上操作
rsync -avz /data/ rsync_backup@172.16.1.41::nfsbackup/ –password-file=/etc/rsync.password

客户端操作
安装
yum install rsync
配置
vim /etc/rsync.password
chmod 600 /etc/rsync.password
推送测试
rsync -avz -P hello.txt rsync_backup@172.33.0.83::nfsbackup –password-file=/etc/rsync.password
