搭建环境
安装基础软件
yum install lrzsz nmap tree dos2unix nc -y
打补丁
yum update或者yum up grade
安装
yum install nfs-utils rpcbind -y
启动RPC
systemctl start rpcbind.service
查看RPc中的启动服务
rpcinfo -p localhost
启动nfs
systemctl start nfs.service
查看rpc中的服务
rpcinfo -p localhost
RPC端口：111
查看端口
netstat -lntup|grep rpcbind
lsof -i :rpcbind
查看rpc中的服务
rpcinfo -p localhost
启动nfs
/etc/init.d/nfs

nfs配置文件
/etc/exports
rsync配置文件
/etc/rsyncd.conf

每个程序都有一个虚拟用户，nfs虚拟用户是:
nfsnobody
创建nfs目录
mkdir /data -p
赋予nfs程序能读写权限
chown nfsnobody.nfsnobody /data
编辑nfs服务器配置文件
vim /etc/exports
注意些注释
#sharc /data by oldboy for bingbing at 20180518
/data 172.16.1.0/24(rw,sync)#sync是写到磁盘，如果不写则写到内存,async异步写入
以上配置需要注意
1.先启动rpcbind
2.后启动nfs
3.查看rpc状态
rpcinfo -p localhost

重启nfs
/etc/init.d/nfs reload 平滑重启，已经到达服务器的请求处理完毕，没有到达的则就屏蔽掉
/etc/init.d/nfs reload ===pxportfs -rv

1.下载工具
2.解压工具，规整目录
mkdir /application -p
mv sersync2.5.4_64bit_binary_stable_final.tar.gz /application/
cd application/
tar zxvf sersync2.5.4_64bit_binary_stable_final.tar.gz
rm -f sersync2.5.4_64bit_binary_stable_final.tar.gz
cp -a GNU-Linux-x86 sersync
cd sersync/
mkdir    conf logs bin
mv /bin/sersync2 bin/
mv confxml.xml conf/
mv bin/sersync2 bin/sersync 


编辑配置文件
<?xml version="1.0" encoding="ISO-8859-1"?>
<head version="2.5">
    <host hostip="localhost" port="8008"></host>
    <debug start="false"/>
    <fileSystem xfs="false"/>
    <filter start="false">
    <exclude expression="(.*)\.svn"></exclude>
    <exclude expression="(.*)\.gz"></exclude>
    <exclude expression="^info/*"></exclude>
    <exclude expression="^static/*"></exclude>
    </filter>
    <inotify>
    <delete start="true"/>
    <createFolder start="true"/>
    <createFile start="false"/>
    <closeWrite start="true"/>
    <moveFrom start="true"/>
    <moveTo start="true"/>
    <attrib start="false"/>
    <modify start="false"/>
    </inotify>

    <sersync>
    <localpath watch="/data">    #同步的目录
        <remote ip="172.16.1.41" name="nfsbackup"/> #目标IP和模块
        <!--<remote ip="192.168.8.39" name="tongbu"/>-->
        <!--<remote ip="192.168.8.40" name="tongbu"/>-->
    </localpath>
    <rsync>
        <commonParams params="-avz"/>    #Rsync同步参数
        <auth start="true" users="rsync_backup" passwordfile="/etc/rsync.password"/>#用户名密码
        <userDefinedPort start="false" port="874"/><!-- port=874 -->
        <timeout start="true" time="100"/><!-- timeout=100 -->##超时时间
        <ssh start="false"/>
    </rsync>
    <failLog path="/application/logs/rsync_fail_log.sh" timeToExecute="60"/><!--default every 60mins execute once-->#日志存储位置
    <crontab start="false" schedule="600"><!--600mins-->
        <crontabfilter start="false">
        <exclude expression="*.php"></exclude>
        <exclude expression="info/*"></exclude>
        </crontabfilter>
    </crontab>
    <plugin start="false" name="command"/>
    </sersync>

    <plugin name="command">
    <param prefix="/bin/sh" suffix="" ignoreError="true"/>    <!--prefix /opt/tongbu/mmm.sh suffix-->
    <filter start="false">
        <include expression="(.*)\.php"/>
        <include expression="(.*)\.sh"/>
    </filter>
    </plugin>

    <plugin name="socket">
    <localpath watch="/opt/tongbu">
        <deshost ip="192.168.138.20" port="8009"/>
    </localpath>
    </plugin>
    <plugin name="refreshCDN">
